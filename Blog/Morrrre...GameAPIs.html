# sendChat

  • 参数及类型：
    • content:string (opens new window)
    • targetuin:number (opens new window)发给uin为targetuin的玩家，1表示发给所有玩家
  • 返回值及类型：
    • ErrorCode (opens new window).OK
  • 该方法的主要作用：发送聊天消息
  • 具体使用案例如下：
ClientCurGame:sendChat(GetS(8024), 1)

# _getCurDir

    • @1 :string: 返回存档路径(data/...)

SSClassIO:_getCurDir(owid, modpacketid)

# coordDivBlock

  • 参数及类型：
    • pos:number (opens new window) 坐标值
  • 返回值及类型：
    • num:number (opens new window):
  • 该方法的主要作用：coord坐标
  • 具体使用案例如下：
function BucketEmpty_AutoUse(world, grid, posX, posY, posZ, dirX, dirY, dirZ, dir, shooterObjIdHigh, shooterObjIdLow)
	local x = world:coordDivBlock(posX);
	local y = world:coordDivBlock(posY);
	local z = world:coordDivBlock(posZ);
	local id = world:getBlockID(x,y,z)

	newitem = 0
	if world:getBlockData(x,y,z) == 0 then
		if id==BLOCK_STILL_WATER or id==BLOCK_FLOW_WATER then newitem = ITEM_BUCKET_WATER
		elseif id==BLOCK_STILL_LAVA or id==BLOCK_FLOW_LAVA then newitem = ITEM_BUCKET_LAVA end
	end

	if newitem > 0 then
		world:setBlockAll(x,y,z, 0, 0)
		if grid:getNum() == 1 then
			grid:setItem(newitem, 1);
			return -1;
		else
			local itemid = grid:getItemID();
			world:getActorMgr():throwItemMotion(posX, posY, posZ, itemid, dirX, dirY, dirZ, shooterObjIdHigh, shooterObjIdLow);
			return 1;
		end
	end

	return 0
end

# dev_developer_info

  • 参数及类型：
  • 返回值及类型：
    • @1 :string: 返回存档路径(data/...)
  • 该方法的主要作用：检测原始地图作者是否是开发者认证账号(主机有效)(回调方式)
  • 具体使用案例如下：
	checkRealOwnerIsDeveloperCallback = function(self, funcCallback, ...)
		if not funcCallback then
			return
		end

		threadpool:work(function(...)
			local realowneruin = GetRealOwnerUin()
			if not realowneruin then
				return
			end

			local isDeveloper = AccountManager and AccountManager.dev_developer_info and AccountManager:dev_developer_info(realowneruin) or ErrorCode.FAILED
			if isDeveloper == ErrorCode.OK or  ClientMgr:isPureServer() == true then
				funcCallback(...)
			else
				--MessageBox(4, "") -- 提示不是开发者
			end
		end, ...)    

    AccountManager:dev_developer_info(realowneruin)
